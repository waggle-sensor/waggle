6 void form3_hex_string_to_hex() // Hex to hex: form3
7 {
8  Temp_ulong[0] = 0x000000;
9  Temp_ulong[1] = 0x000000;
10 
11  for (int i = 0; i < VAL_NUM_ID; i++)
12  {
13  if (VAL[i] > '9') // letters
14  VAL[i] = VAL[i] - 'a' + 0x0a;
15  else // numbers
16  VAL[i] = VAL[i] - '0';
17 
18  if (i < VAL_NUM_ID / 2)
19  Temp_ulong[0] = Temp_ulong[0] | (VAL[i] << (VAL_NUM_ID / 2 - i - 1) * 4);
20  else
21  Temp_ulong[1] = Temp_ulong[1] | (VAL[i] << (VAL_NUM_ID - i - 1) * 4);
22  }
23 
24  Temp_byte[0] = (Temp_ulong[0] & 0xFF0000) >> 16;
25  Temp_byte[1] = (Temp_ulong[0] & 0x00FF00) >> 8;
26  Temp_byte[2] = (Temp_ulong[0] & 0x0000FF);
27  Temp_byte[3] = (Temp_ulong[1] & 0xFF0000) >> 16;
28  Temp_byte[4] = (Temp_ulong[1] & 0x00FF00) >> 8;
29  Temp_byte[5] = (Temp_ulong[1] & 0x0000FF);
30  format3(Temp_byte);
31 }

119 void chemsense_acquire()
120 {
121  if (Serial3.available() > 0)
122  {
123  count_chem = count;
124  while (flag_CHEM_WHILE == true && count < count_chem + 4)
125  {
126  INPUT_BYTE = Serial3.read(); //read the incoming byte
127 
128  if (INPUT_BYTE >= 'a' && INPUT_BYTE <= 'z' ||
129  INPUT_BYTE >= 'A' && INPUT_BYTE <= 'Z' ||
130  INPUT_BYTE >= '0' && INPUT_BYTE <= '9' ||
131  INPUT_BYTE == '-')
132  {
133  if (!flag_KEY)
134  {
135  if (KEY_NUM_ID < 3)
136  {
137  KEY[KEY_NUM_ID] = INPUT_BYTE;
138  KEY_NUM_ID++;
139  }
140  else
141  KEY_NUM_ID = 0;
142  }
143  else
144  {
145  if (VAL_NUM_ID < 12)
146  {
147  VAL[VAL_NUM_ID] = INPUT_BYTE;
148  VAL_NUM_ID++;
149  }
150  else
151  {
152  KEY_NUM_ID = 0;
153  VAL_NUM_ID = 0;
154  flag_KEY = false;
155  }
156  }
157  }
158  else if (INPUT_BYTE == '=')
159  {
160  flag_KEY = true;
161  }
162  else if (INPUT_BYTE == '\r' || INPUT_BYTE == ' ' || INPUT_BYTE == '\n')
163  {
164  //KEY[KEY_NUM_ID] = '\0';
165  //VAL[VAL_NUM_ID] = '\0';
166  Carrier();
167  KEY_NUM_ID = 0;
168  VAL_NUM_ID = 0;
169  }
170  }
171  }
172 }
